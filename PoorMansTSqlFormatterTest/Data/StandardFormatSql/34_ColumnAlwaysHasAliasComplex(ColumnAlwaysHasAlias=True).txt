-- Test: columnAlwaysHasAlias with complex expressions and multiple SELECT blocks
SELECT c.CustomerID AS CustomerID
    ,c.Name AS Name
    ,SUM(o.Amount) AS Total
    ,AVG(o.Amount) AS ColumnAlias_1
FROM Customers c
LEFT JOIN Orders o ON c.CustomerID = o.CustomerID
GROUP BY c.CustomerID
    ,c.Name

SELECT e.*
    ,e.Salary * 1.15 AS Bonus
FROM Employees e
WHERE e.Active = 1
